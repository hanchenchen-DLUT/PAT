1. `Inorder`+`Level`+`BST`

   ```c++
   #include<iostream>
   #include<algorithm>
   #include<queue>
   #include<string.h>
   using namespace std;
   int l[105];
   int r[105];
   int bst[105];
   int x=0,n;
   int arr[105];
   void inorder(int root){
       if(root>=n||root==-1)return;
       inorder(l[root]);
       bst[root]=arr[x++];
       inorder(r[root]);
   }
   int main(){
       cin>>n;
       memset(l,-1,sizeof(l));
       memset(r,-1,sizeof(r));
       for(int i=0;i<n;i++){
           cin>>l[i]>>r[i];
       }
       for(int i=0;i<n;i++){
           cin>>arr[i];
       }
       sort(arr,arr+n);
       inorder(0);
       queue<int> qe;
       qe.push(0);
       int flag=0;
       while(!qe.empty()){
           if(l[qe.front()]!=-1)qe.push(l[qe.front()]);
           if(r[qe.front()]!=-1)qe.push(r[qe.front()]);
           if(!flag)flag=1;
           else cout<<" ";
           cout<<bst[qe.front()];
           qe.pop();
       }
       return 0;
   }
   ```
